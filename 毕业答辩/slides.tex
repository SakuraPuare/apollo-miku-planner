% [02] 研究背景 —— 自动驾驶路径规划的三难
% ============================================================
\begin{frame}{研究背景}
  \small 自动驾驶从感知到控制之间，路径规划承担轨迹生成职责，直接决定整车安全与舒适。
  \vspace{0.4em}
  \begin{columns}[T,onlytextwidth]
    \column{0.56\textwidth}
    \begin{block}{多障碍物密集场景的三难}
      \begin{itemize}\setlength\itemsep{0.3em}
        \item \textbf{可行域收缩}\quad{}禁行区域叠加使横向通行空间急剧压缩，甚至降为零
        \item \textbf{决策耦合}\quad{}多障碍物的左右避让方向相互制约，独立决策易产生矛盾方向
        \item \textbf{时间预算紧}\quad{}车载规划周期不超过$100$\,ms，$k$ 个障碍物的$O(2^k)$组合决策须压缩到可解规模
      \end{itemize}
    \end{block}
    \column{0.42\textwidth}
    \begin{exampleblock}{研究平台与切入点}
      \begin{itemize}\setlength\itemsep{0.3em}
        \item \textbf{平台}\quad{}百度 Apollo 11.0 开源
        \item \textbf{流程}\quad{}PublicRoad Planner 路径-速度解耦
        \item \textbf{切入点}\quad{}路径边界决策 \texttt{PathBoundsDecider}
      \end{itemize}
    \end{exampleblock}
  \end{columns}
\end{frame}

      \end{enumerate}
    \end{block}
    \small 三个症状分别对应安全距离、决策粒度、时空协调三条维度。

      \begin{itemize}\setlength\itemsep{0.2em}
        \item $f_{\mathrm{TTC}}$\quad{}碰撞时间映射
        \item $f_{\mathrm{overlap}}$\quad{}横向重叠率
        \item $f_{\mathrm{vel}}$\quad{}相对速度Sigmoid

    \begin{block}{量化改造}
      \footnotesize
      $d_{buf,i} = d_{buf,\min} + (d_{buf,\max}-d_{buf,\min})\cdot \Theta_i,\quad \delta_i = W_{ego}/2 + d_{buf,i}$
      \par\vspace{0.15em}
      额外缓冲$d_{buf,i}\in[\DeltaIntervalLow,\DeltaIntervalHigh]$\,m，取代Baseline一刀切的$d_{buf}\!=\!\DeltaBaseline$\,m；
      5个因子加权合成$\Theta_i\!\in\![0,1]$，锥桶$\Theta_i$低至$\sim$0.2，行人$\Theta_i$高至$\sim$0.9。
      \par\vspace{0.15em}
      \textbf{鲁棒性}\quad{}权重$\pm\SensPerturbPct$\%扰动下3/4场景翻转率\SensOneMcFlipRate\%，最坏相对偏差\SensWorstRelDev\%。
    \end{block}
  \end{columns}
\end{frame}

% [09] 创新二：组级最优通行带 —— 方法
% ============================================================
\begin{frame}{创新二\quad{}组级最优通行带}
  \vspace{-0.6em}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \adjustbox{max width=\linewidth,max totalheight=0.65\textheight}{\input{../图片/fig_corridor_gen_a}}
    \column{0.50\textwidth}
    \begin{block}{复杂度由指数降至对数线性}
      \footnotesize
      \begin{itemize}\setlength\itemsep{0.2em}
        \item \textbf{引理}\quad{}最优分割在 $u_i$ 排序下连续
        \item \textbf{定理}\quad{}越界预处理后，组内按 $u_i$ 升序的 $k+1$ 个候选间隙中最大者即最优 $\mathbf{d}^*$
        \item 搜索空间\quad{}\ComplexityOldBase$\to$\ComplexityNewBase
        \item 组规模上限 $k_{\max}\!=\!\GroupKmax$，超限回退 Baseline 逐障碍物策略
      \end{itemize}
    \end{block}
    \begin{block}{单次排序求解}
      \footnotesize
      扫描线按 $s_i^-$ 升序得连通组，组内按 $u_i$ 升序得 $k+1$ 个候选间隙，取宽度最大者即 $\mathbf{d}^*$。
      \par\vspace{0.15em}
      \textbf{实测佐证}\quad{}P4 密集施工下 $k\!=\!\GroupCThreeK$，Baseline 需 $2^{\GroupCThreeK}\!\approx\!\GroupCThreeCombCount$ 万次组合判断，MIKU 按排序一次给出最优解；w/o 组件 C2--C3 变体仅走 \AblationMTwoScFourSEndPct\% 路程即降级停车，MIKU 全通。
    \end{block}
  \end{columns}
\end{frame}

% [19] 主要结论
% ============================================================
\begin{frame}{主要结论}
  \begin{center}\small\colorbox{bglight}{\;一组符号$(u_i,v_i,\delta,\tau)$，三项改进，一步\,\texttt{PathBoundsDecider}\;}\end{center}
  \vspace{-0.2em}
  \footnotesize
  \begin{itemize}\setlength\itemsep{0.35em}
    \item \textbf{核心}\quad{}可比 C1--C4 场景 MIKU 与 Baseline 通行率持平，轨迹平顺性与 QP 开销有改善；压力 P1--P4 场景通行率由 \MPressureBaselinePassCount/\MPressureScenarioTotal 提升至 \MPressureMikuPassCount/\MPressureScenarioTotal
    \item \textbf{方法}\quad{}三项改进共用 $(u_i,v_i,\delta,\tau)$ 一组符号；源码层只动 \texttt{PathBoundsDecider} 一步，PJPO/PJSO/Control 链路零改动
    \item \textbf{效率}\quad{}整体平均速度提升 \MSummarySummaryVImprovement\%；可比场景 QP 总耗时 \MComparableBaselineQpTotal$\to$\MComparableMikuQpTotal\,ms
    \item \textbf{计算}\quad{}可比 QP 降 65.6\%，整体仅多付 \MSummarySummaryExtraMs\,ms；压力场景 Baseline 降级不作效率比较
    \item \textbf{消融}\quad{}M1--M3 关任一路径组件即在 P1--P4 失败；w/o 组件 C5 仅差 $\AblationMarginMFourMFive$ 分，路径四组件互补
    \item \textbf{鲁棒}\quad{}$\pm\SensPerturbPct$\% 权重 $\times$ \SensTrialCount 次蒙特卡洛，P1--P3 翻转率 \SensOneMcFlipRate\%、P4 由走廊兜底，最坏相对偏差 \SensWorstRelDev\%
    \item \textbf{边界}\quad{}当前限于数值仿真与 Dreamview 在环；实车感知、延迟与嵌入式算力映射有待评估
  \end{itemize}
\end{frame}
