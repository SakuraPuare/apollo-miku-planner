      \begin{itemize}\setlength\itemsep{0.3em}

    \end{block}

    \begin{exampleblock}{研究基座}
      \begin{itemize}\setlength\itemsep{0.3em}
        \item 百度Apollo 11.0 开源平台
        \item PublicRoad Planner 的PJPO路径QP链路
        \item 路径边界决策模块PathBoundsDecider
      \end{itemize}
    \end{exampleblock}
  \end{columns}
  \vspace{0.3em}
  \small 自动驾驶从感知到控制之间，路径规划承担轨迹生成职责，直接决定整车安全与舒适。
\end{frame}

      \begin{itemize}\setlength\itemsep{0.2em}
        \item $T_{\mathrm{TTC}}$\quad{}碰撞时间倒数
        \item $T_{\mathrm{ov}}$\quad{}横向重叠率
        \item $T_{v_{rel}}$\quad{}相对速度方向
        \item $T_{\mathrm{type}}$\quad{}障碍物类别先验
        \item $T_{\rho}$\quad{}交互密度
      \end{itemize}
    \end{block}
    \begin{block}{量化改造}
      \footnotesize
      $\delta_i = \delta_{\min} + (\delta_{\max}-\delta_{\min})\cdot T_i$
      \par\vspace{0.15em}
      裕度区间$\delta_i \in [\DeltaIntervalLow, \DeltaIntervalHigh]$\,m，取代Baseline一刀切的\DeltaBaseline\,m；
      5个因子张成评分空间，锥桶$T_i$低落至$\sim$0.2，行人$T_i$高至$\sim$0.9。
      \par\vspace{0.15em}
      \textbf{鲁棒性}\quad{}权重$\pm\SensPerturbPct$\%扰动下3/4场景翻转率\SensOneMcFlipRate\%，最坏相对偏差仅\SensWorstRelDev\%。
    \end{block}
  \end{columns}

% [09] 创新二：组级最优通行带 —— 方法
% ============================================================
\begin{frame}{创新二\quad{}组级最优通行带}
  \vspace{-0.6em}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \adjustbox{max width=\linewidth,max totalheight=0.65\textheight}{\input{../图片/fig_corridor_gen_a}}
    \column{0.50\textwidth}
    \begin{block}{复杂度从指数压到对数线性}
      \footnotesize
      \begin{itemize}\setlength\itemsep{0.2em}
        \item \textbf{引理}\quad{}最优分割在横向排序下连续
        \item \textbf{定理}\quad{}组内按$l_{\min}$排序后，$k+1$个候选间隙里的最大者即最优
        \item 搜索空间\quad{}\ComplexityOldBase$\to$\ComplexityNewBase
        \item 组规模上限$k_{\max}\!=\!\GroupKmax$，超限回退Baseline逐障碍物策略
      \end{itemize}
    \end{block}
    \begin{block}{一步出解}
      \scriptsize
      扫描线得连通组$\to$组内按$l_{\min}$排序得$k+1$个间隙$\to$取宽度最大间隙即$\mathbf{d}^*$。C3密集施工下$k\!=\!\GroupCThreeK$，Baseline需$2^{\GroupCThreeK}\!\approx\!\GroupCThreeCombCount$万次组合判断，MIKU按排序一步出解。
      \par\vspace{0.15em}
      \textbf{实测佐证}\quad{}C2行人加停车QP总耗时\MTwoBaselineQpTotal$\to$\MTwoMikuQpTotal\,ms，约24倍加速。
    \end{block}
  \end{columns}
\end{frame}

  \begin{block}{结论}
    \footnotesize
    \begin{itemize}\setlength\itemsep{0.15em}
      \item 单项改进（M1--M3）因问题维度不匹配，评分天花板被\AblationFailedScoreCeil 分的失败惩罚压制
      \item M4留一不加仍达\AblationScoreMFour 分，与全量M5仅相差\AblationMarginMFourMFive 分，说明\textbf{三项改进高度互补}
      \item 组合成MIKU全量（M5）即达评分上限\AblationScoreMFive，\textbf{缺一则退化}
    \end{itemize}
  \end{block}

\begin{frame}{主要结论}
  \footnotesize
  \begin{itemize}\setlength\itemsep{0.35em}
    \item 针对Apollo PathBoundsDecider三项源码层局限，提出多障碍物交互运动学统一规划算法MIKU，以差异化$\delta_i$、组级最优$\mathbf{d}^*$、时空走廊$\tau(s)$三条改进共用同一组$(u_i, v_i, \delta, \tau)$数学记号
    \item \textbf{通过率}\quad{}压力场景\MPressureBaselinePassCount/\MPressureScenarioTotal$\to$\MPressureMikuPassCount/\MPressureScenarioTotal，整体\MSummaryBaselinePassCount/\MScenarioTotal$\to$\MSummaryMikuPassCount/\MScenarioTotal
    \item \textbf{效率}\quad{}整体平均速度提升\MSummarySummaryVImprovement\%；压力场景平均速度\MPressureBaselineAvgV$\to$\MPressureMikuAvgV\,m/s
    \item \textbf{计算}\quad{}压力场景QP总耗时\MPressureBaselineQpTotal$\to$\MPressureMikuQpTotal\,ms（降84.3\%），可比场景\MComparableBaselineQpTotal$\to$\MComparableMikuQpTotal\,ms（降66.6\%），仅多付\MSummarySummaryExtraMs\,ms平均开销
    \item \textbf{消融}\quad{}M0原始\AblationScoreMZero 分$\to$M5全量\AblationScoreMFive 分；留一不加M4仅\AblationScoreMFour 分（差\AblationMarginMFourMFive），证三项改进高度互补
    \item \textbf{鲁棒}\quad{}威胁度权重$\pm\SensPerturbPct$\% $\times$ \SensTrialCount 次蒙特卡洛，3/4场景翻转率\SensOneMcFlipRate\%，最坏相对偏差\SensWorstRelDev\%
    \item 已完成C++增量接入，保持Apollo数据流与QP求解链路不变，下游Control模块无需修改
  \end{itemize}
\end{frame}
