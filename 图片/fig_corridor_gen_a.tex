% 子图 (a)：单 s 截面横向间隙判定（教学起点）
% 选 s = s^* = 3.25 卡在静态障碍 A 中部，沿 l 轴展示 max-gap 选 p^* 过程
% A 静态：l ∈ [-1.2, 0.3]，u_A=-1.2, v_A=0.3
% g_0 = u_A - l_road^- = -1.2 - (-2) = 0.8 < W_ego = 1.5, no
% g_1 = l_road^+ - v_A =  2  - 0.3  = 1.7 >= W_ego = 1.5, ok, p^* = 1
% 自车宽度 W_ego = 1.5 嵌入右侧间隙 g_1
\begin{tikzpicture}[scale=1.0, transform shape]
  \useasboundingbox (-3.0, -2.1) rectangle (3.0, 2.4);

  % ===== 标题 =====
  \node[font=\footnotesize\bfseries] at (0, 2.10) {$s = s^\star = 3.25$ 横向截面};

  % ===== 道路两侧边界（竖线）=====
  \draw[gray!75, thick] (-2, -0.55) -- (-2, 1.55);
  \draw[gray!75, thick] ( 2, -0.55) -- ( 2, 1.55);
  \node[gray!60!black, font=\tiny, anchor=south] at (-2, 1.55) {$l_{road}^{-}$};
  \node[gray!60!black, font=\tiny, anchor=south] at ( 2, 1.55) {$l_{road}^{+}$};

  % ===== 不通行间隙 g_0（左）：浅红淡色带 =====
  \fill[dangerred!18] (-2, 0.05) rectangle (-1.2, 1.05);
  \draw[dangerred!60, thick, |-|] (-2, 1.30) -- (-1.2, 1.30);
  \node[dangerred!75!black, font=\scriptsize] at (-1.6, 1.55) {$g_0 = 0.8$};
  \node[dangerred, font=\footnotesize\bfseries] at (-1.6, 0.55) {$\times$};

  % ===== 通行间隙 g_1（右）：lightgreen 色带 =====
  \fill[lightgreen!50] (0.3, 0.05) rectangle (2, 1.05);
  \draw[lightgreen!55!black, very thick, |-|] (0.3, 1.30) -- (2, 1.30);
  \node[lightgreen!45!black, font=\scriptsize] at (1.15, 1.55) {$g_1 = 1.7$};
  \node[lightgreen!45!black, font=\footnotesize\bfseries] at (1.65, 0.55) {$\checkmark$};

  % ===== 障碍 A 的 l 投影矩形 =====
  \fill[dangerred!75, draw=dangerred, thick] (-1.2, 0.05) rectangle (0.3, 1.05);
  \node[font=\scriptsize\bfseries, white] at (-0.45, 0.55) {A};

  % ===== 自车（W_ego=1.5）嵌入 g_1，居中于 [0.35, 1.85] =====
  \fill[deepblue, draw=deepblue, thick, fill opacity=0.55]
    (0.35, 0.20) rectangle (1.85, 0.90);
  \node[deepblue!20!black, font=\scriptsize\bfseries] at (1.10, 0.55) {自车};
  \draw[deepblue, thick, |-|] (0.35, -0.10) -- (1.85, -0.10);
  \node[deepblue, font=\scriptsize, anchor=north] at (1.10, -0.13) {$W_{ego} = 1.5$};

  % ===== l 轴（在道路下方）=====
  \draw[-Stealth, thick, black!75] (-2.65, -0.55) -- (2.7, -0.55)
    node[right, font=\footnotesize] {$l$};
  \foreach \x in {-2, -1, 0, 1, 2} {
    \draw[thin, black!75] (\x, -0.55) -- (\x, -0.45);
    \node[font=\tiny, below=1pt] at (\x, -0.55) {$\x$};
  }

  % ===== u_A, v_A 端点标注 =====
  \draw[dangerred, thin, dotted] (-1.2, 0.05) -- (-1.2, -0.85);
  \draw[dangerred, thin, dotted] ( 0.3, 0.05) -- ( 0.3, -0.85);
  \node[dangerred, font=\tiny] at (-1.2, -0.95) {$u_A = -1.2$};
  \node[dangerred, font=\tiny] at ( 0.3, -0.95) {$v_A =  0.3$};

  % ===== 底部判定一行 =====
  \node[font=\scriptsize, gray!55!black] at (0, -1.55)
    {$g^\star = \max(g_0, g_1) = 1.7 \geq W_{ego}$，故 $p^\star = 1$ 通行};
  \node[font=\tiny, lightgreen!45!black] at (0, -1.92)
    {该截面对应子图 (b) 中 A 段水平平台 $1.7$};
\end{tikzpicture}
