% Project-local font setup for XeLaTeX.
% Callers may set \ProjectFontDir before \input'ing this file.
\providecommand{\ProjectFontDir}{../fonts}

\setmainfont[
  Path = \ProjectFontDir/tex-gyre/,
  Extension = .otf,
  UprightFont = *-regular,
  BoldFont = *-bold,
  ItalicFont = *-italic,
  BoldItalicFont = *-bolditalic,
  Ligatures = TeX
]{texgyretermes}

]{FandolSong-Regular}
