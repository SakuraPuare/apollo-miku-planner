\section{研究背景与意义}
% ==============================================================

%% ---- 幻灯片 3：研究背景（一） ----
\begin{frame}[shrink=3]{研究背景（一）：自动驾驶与Apollo平台}
  \begin{columns}[T]
    \column{0.45\textwidth}
    \begin{block}{自动驾驶发展现状}
      \begin{itemize}
        \item 人工智能、传感器、计算机视觉技术快速发展
        \item 自动驾驶从实验室走向商业化应用阶段
        \item \alert{路径规划}是决定行驶轨迹的安全关键模块
      \end{itemize}
    \end{block}
    \vspace{0.1em}
    \begin{block}{百度Apollo平台}
      \begin{itemize}
        \item 全球领先的\textbf{开源}自动驾驶平台
        \item 完整的感知--预测--规划--控制技术栈
        \item \textbf{Planning模块}：基于凸优化的局部轨迹规划
      \end{itemize}
    \end{block}

    \column{0.45\textwidth}
    \begin{alertblock}{城市多障碍物场景的复杂性}
      \begin{itemize}
        \item \textbf{动态障碍物}：机动车、行人、非机动车，数量多、状态多变
        \item \textbf{静态障碍物}：施工区域、临时路障、违停车辆
        \item 传统规划算法：复杂度高、避障策略过于保守
      \end{itemize}
    \end{alertblock}
    \begin{exampleblock}{研究聚焦}
      Apollo Planning模块在\alert{多障碍物密集场景}下的性能优化
    \end{exampleblock}
  \end{columns}
\end{frame}

%% ---- 幻灯片 4：研究背景（二）+ ST图示意 ----
\begin{frame}{研究背景（二）：多障碍物场景三大挑战}
  \begin{columns}[T]
    \column{0.56\textwidth}
    \begin{columns}[c]
      \column{0.3\textwidth}
      \begin{alertblock}{\centering 挑战一\\可行域压缩}
        \centering\small
        ST边界独立\\扩展安全距离\\$\Rightarrow$边界叠加\\$\Rightarrow$可行域趋零
      \end{alertblock}
      \column{0.3\textwidth}
      \begin{alertblock}{\centering 挑战二\\决策耦合}
        \centering\small
        PathDecider\\逐障碍物贪心\\$\Rightarrow$缺全局最优\\$\Rightarrow$STOP频发
      \end{alertblock}
      \column{0.3\textwidth}
      \begin{alertblock}{\centering 挑战三\\实时约束}
        \centering\small
        计算量随障碍\\物数量增长\\$\Rightarrow$耗时上升\\$\Rightarrow$实时性差
      \end{alertblock}
    \end{columns}
    \vspace{0.4em}
    \begin{block}{研究动机}
      \small Apollo在多障碍物密集场景下\alert{保守性过强、通行效率低}，规划频繁失败，亟需针对性改进
    \end{block}

    \column{0.42\textwidth}
    \centering
    {\small\textbf{ST边界叠加问题直观示意}}\\[3pt]
    \input{figures/fig_st_problem.tex}\\[2pt]
    \footnotesize{\textit{多障碍物安全缓冲叠加$\Rightarrow$可行域极度压缩}}
  \end{columns}
\end{frame}

        \item 识别凸优化$+$ST图方法的多障碍物性能瓶颈
        \item 为路径规划改进研究提供分析方法参考

    \end{block}
    \vspace{0.3em}

        \item 提升多障碍物密集场景通行效率
        \item 减少不必要停车，改善乘坐舒适性
      \end{itemize}
    \end{block}

    \column{0.52\textwidth}
    \begin{exampleblock}{研究目标（四步递进）}

              与核心算法实现机制
        \item \textbf{识别问题}：ST边界重叠、\\
              PathDecider决策保守、场景切换滞后
        \item \textbf{设计改进算法}：基于动态权重\\
              的多障碍物处理与时空走廊优化
        \item \textbf{仿真验证}：多类型场景下与\\
              Apollo原算法的定量对比分析
      \end{enumerate}
    \end{exampleblock}
  \end{columns}
\end{frame}

\section{国内外研究现状}
% ==============================================================

%% ---- 幻灯片 6：国外研究现状 ----
\begin{frame}{国外研究现状：四类主流路径规划方法}
  \begin{columns}[T]
    \column{0.45\textwidth}
    \begin{block}{\cn{1} 图搜索方法}
      \begin{itemize}
        \item \textbf{Hybrid A*} [Dolgov, 2010]：连续空间$+$离散搜索，满足运动学约束
        \item Apollo应用：Open Space Planner（泊车/掉头）
      \end{itemize}
    \end{block}
    \vspace{0.1em}
    \begin{block}{\cn{2} 采样规划方法}
      \begin{itemize}
        \item \textbf{Frenet轨迹规划} [Werling, 2010]：横纵向独立采样，评估筛选
        \item Apollo应用：Lattice Planner（城市道路）
      \end{itemize}
    \end{block}

    \column{0.45\textwidth}
    \begin{block}{\cn{3} 凸优化方法}
      \begin{itemize}
        \item \textbf{QP二次规划} [Chen, 2019]：OSQP求解器高效求解
        \item Apollo应用：PublicRoadPlanner路径-速度解耦优化\\
              （\alert{本文改进核心对象}）
      \end{itemize}
    \end{block}
    \vspace{0.1em}
    \begin{block}{\cn{4} 学习型方法}
      \begin{itemize}
        \item 深度/强化学习端到端规划
        \item 可解释性差，安全验证困难，工业部署受限
      \end{itemize}
    \end{block}
  \end{columns}
\end{frame}

%% ---- 幻灯片 7：国内研究现状与切入点 ----
\begin{frame}{国内研究现状与研究切入点}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{国内研究进展}
    \begin{itemize}
      \item Apollo源码分析与特定场景（泊车/高速）优化
      \item \textbf{时空走廊轨迹规划} [王鹏, 2023]\\
            构建安全通道简化多障碍物约束处理
      \item \textbf{Transformer预测性运动规划} [张华, 2024]\\
            提升动态场景下预测与规划适应性
      \item \textbf{横纵向协同避障研究} [陈志远, 2024]\\
            复杂道路环境下横纵向联合优化
    \end{itemize}

    \column{0.49\textwidth}
    \begin{alertblock}{现有研究不足（本文切入点）}
      \begin{enumerate}
        \item \textbf{ST边界优化不足}\\
              多障碍物独立边界叠加\\
              $\rightarrow$ 速度规划可行域过度压缩
        \item \textbf{障碍物交互建模有限}\\
              独立处理各障碍物，\\
              缺乏有效的交互关系量化与分级
        \item \textbf{场景切换机制不灵活}\\
              Scenario-based架构切换滞后，\\
              密集交通场景响应速度受限
      \end{enumerate}
    \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}{主要研究内容（四个方面）}
  \begin{columns}[T]
    \column{0.45\textwidth}
    \begin{block}{（1）Apollo Planning模块深入分析}
      \begin{itemize}
        \item Routing与Planning数据交互机制
        \item Scenario-Stage-Task三级管理架构
        \item 路径-速度解耦两阶段优化（路径QP$+$速度QP）
        \item PathDecider / SpeedBoundsDecider实现细节
      \end{itemize}
    \end{block}
    \vspace{0.1em}
    \begin{block}{（2）多障碍物场景问题识别}
      \begin{itemize}
        \item ST边界重叠机制的定量分析
        \item PathDecider保守决策根本原因溯源
        \item 场景切换滞后问题定位
      \end{itemize}
    \end{block}

    \column{0.45\textwidth}
    \begin{block}{（3）\alert{改进算法设计与实现（核心）}}
      \begin{itemize}
        \item 多因素威胁评估模型（5维量化）
        \item 分级障碍物决策框架（3级差异处理）
        \item 时空可通行走廊算法（扫描线$+$回溯）
      \end{itemize}
    \end{block}
    \vspace{0.05em}
    \begin{block}{（4）算法验证与对比分析}
      \begin{itemize}
        \item 四类仿真场景：纵向/横向/动静混合/路口
        \item 与Apollo原算法对比：成功率/速度/停车次数
      \end{itemize}
    \end{block}
  \end{columns}
\end{frame}

%% ---- 幻灯片 10：改进算法概览 ----
\begin{frame}{改进算法概览}
  \begin{columns}[T]
    \column{0.45\textwidth}
    \begin{block}{多因素威胁评估模型}
      五维加权评分：
      \[
        T_{\mathrm{score}} = \sum_{i=1}^{5} w_i \cdot f_i
      \]
      \begin{itemize}
        \item $f_1$：TTC（碰撞时间）
        \item $f_2$：横向重叠率
        \item $f_3$：相对速度分量
        \item $f_4$：障碍物类型权重
        \item $f_5$：交互密度因子
      \end{itemize}
    \end{block}

    \column{0.45\textwidth}
    \begin{block}{分级决策框架}
      \small
      \begin{tabular}{@{}lll@{}}
        \toprule
        级别 & 威胁程度 & 处理策略 \\
        \midrule
        \alert{CRITICAL} & 高 & 优先，严格约束 \\
        IMPORTANT        & 中 & 标准处理 \\
        MINOR            & 低 & 简化/忽略 \\
        \bottomrule
      \end{tabular}
    \end{block}
    \vspace{0.4em}
    \begin{block}{时空可通行走廊（核心创新）}
      \begin{itemize}
        \item 扫描线法构造连通时空走廊
        \item 回溯机制确保走廊端到端连通性
        \item 以\alert{单一线性约束替代}多个\\
              独立ST边界叠加约束
        \item 从根本上解决可行域压缩问题
      \end{itemize}
    \end{block}
  \end{columns}
\end{frame}
